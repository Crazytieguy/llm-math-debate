Problem:
Find the eigenvectors of the following matrix:
$\left(
\begin{array}{cc}
 7 & 5 \\
 -10 & 3 \\
\end{array}
\right)$.
Answer:
\begin{array}{l}
 
\begin{array}{l}
 \text{Find all the eigenvalues and eigenvectors of the matrix }M: \\
 M=\left(
\begin{array}{cc}
 7 & 5 \\
 -10 & 3 \\
\end{array}
\right) \\
\end{array}
 \\
\hline
 
\begin{array}{l}
 \text{Find }\lambda \in \mathbb{C} \text{such }\text{that }M v=\lambda  v \text{for }\text{some }\text{nonzero }\text{vector }v: \\
 M v=\lambda  v \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Rewrite }\text{the }\text{equation }M v=\lambda  v \text{as }(M-\mathbb{I} \lambda ) v=0: \\
 (M-\mathbb{I} \lambda ) v=0 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Find }\text{all }\lambda  \text{such }\text{that }| M-\mathbb{I} \lambda | =0: \\
 | M-\mathbb{I} \lambda | =0 \\
\end{array}
 \\
 
\begin{array}{l}
 
\begin{array}{lll}
 M-\mathbb{I} \lambda  & = & \fbox{$\left(
\begin{array}{cc}
 7 & 5 \\
 -10 & 3 \\
\end{array}
\right)$}-\fbox{$\left(
\begin{array}{cc}
 1 & 0 \\
 0 & 1 \\
\end{array}
\right)$} \lambda  \\
 \text{} & = & \left(
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right) \\
\end{array}
 \\
 \left| 
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right| =0 \\
\end{array}
 \\
 
\begin{array}{l}
 \left| 
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right| =\lambda ^2-10 \lambda +71: \\
 \lambda ^2-10 \lambda +71=0 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Subtract }\text{71 }\text{from }\text{both }\text{sides}: \\
 \lambda ^2-10 \lambda =-71 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Add }\text{25 }\text{to }\text{both }\text{sides}: \\
 \lambda ^2-10 \lambda +25=-46 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Write }\text{the }\text{left }\text{hand }\text{side }\text{as }\text{a }\text{square}: \\
 (\lambda -5)^2=-46 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Take }\text{the }\text{square }\text{root }\text{of }\text{both }\text{sides}: \\
 \lambda -5=i \sqrt{46}\text{ or }\lambda -5=-i \sqrt{46} \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Add }5 \text{to }\text{both }\text{sides}: \\
 \fbox{$\lambda =5+i \sqrt{46}$}\text{ or }\lambda -5=-i \sqrt{46} \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Add }5 \text{to }\text{both }\text{sides}: \\
 \lambda =5+i \sqrt{46}\text{ or }\fbox{$\lambda =5-i \sqrt{46}$} \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Find }\text{all }v \text{such }\text{that }(M-\mathbb{I} \lambda ) v=0 \text{for }\text{some }\text{eigenvalue }\lambda : \\
 (M-\mathbb{I} \lambda ) v=0 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Substitute }\left(
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right) \text{for }(M-\mathbb{I} \lambda ): \\
 \fbox{$\left(
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right)$} v=0 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Write }v \text{as }\left(
\begin{array}{c}
 v_1 \\
 v_2 \\
\end{array}
\right) \text{and }0 \text{as }\left(
\begin{array}{c}
 0 \\
 0 \\
\end{array}
\right): \\
 \left(
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right).\fbox{$\left(
\begin{array}{c}
 v_1 \\
 v_2 \\
\end{array}
\right)$}=\fbox{$\left(
\begin{array}{c}
 0 \\
 0 \\
\end{array}
\right)$} \\
\end{array}
 \\
 
\begin{array}{l}
 \text{First, }\text{substitute }5+i \sqrt{46} \text{for }\lambda  \text{in }\text{the }\text{matrix }\left(
\begin{array}{cc}
 7-\lambda  & 5 \\
 -10 & 3-\lambda  \\
\end{array}
\right) \text{and }\text{solve }\text{the }\text{system}: \\
 \fbox{$\left(
\begin{array}{cc}
 2-i \sqrt{46} & 5 \\
 -10 & -i \sqrt{46}-2 \\
\end{array}
\right)$}.\left(
\begin{array}{c}
 v_1 \\
 v_2 \\
\end{array}
\right)=\left(
\begin{array}{c}
 0 \\
 0 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{In }\text{augmented }\text{matrix }\text{form, }\text{the }\text{system }\text{is }\text{written }\text{as}: \\
 \left(
\begin{array}{ccc}
 2-i \sqrt{46} & 5 & 0 \\
 -10 & -i \sqrt{46}-2 & 0 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Swap }\text{row }1 \text{with }\text{row }2: \\
 \left(
\begin{array}{ccc}
 -10 & -i \sqrt{46}-2 & 0 \\
 2-i \sqrt{46} & 5 & 0 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Add }\frac{1}{10} \left(2-i \sqrt{46}\right)\, \times \, \text{(row }1) \text{to }\text{row }2: \\
 \left(
\begin{array}{ccc}
 -10 & -i \sqrt{46}-2 & 0 \\
 0 & 0 & 0 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Divide }\text{row }1 \text{by }-10: \\
 \left(
\begin{array}{ccc}
 1 & \frac{1}{10} \left(2+i \sqrt{46}\right) & 0 \\
 0 & 0 & 0 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Translated }\text{back }\text{to }\text{a }\text{matrix }\text{equation, }\text{the }\text{reduced }\text{system }\left(
\begin{array}{ccc}
 1 & \frac{1}{10} \left(2+i \sqrt{46}\right) & 0 \\
 0 & 0 & 0 \\
\end{array}
\right) \text{is}: \\
 \left(
\begin{array}{cc}
 1 & \frac{1}{10} \left(2+i \sqrt{46}\right) \\
 0 & 0 \\
\end{array}
\right)\left(
\begin{array}{c}
 v_1 \\
 v_2 \\
\end{array}
\right)=\left(
\begin{array}{c}
 0 \\
 0 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{As }\text{a }\text{scalar }\text{equation, }\text{the }\text{system }\left(
\begin{array}{cc}
 1 & \frac{1}{10} \left(2+i \sqrt{46}\right) \\
 0 & 0 \\
\end{array}
\right)\left(
\begin{array}{c}

\end{array}
\right)=\left(
\begin{array}{c}

\end{array}
\right) \text{translates }\text{to}: \\
 v_1+\frac{1}{10} \left(i \sqrt{46}+2\right) v_2=0 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Rewrite }\text{the }\text{equation }\text{as}: \\
 v_1=-\frac{1}{10} \left(2+i \sqrt{46}\right) v_2 \\
\end{array}
 \\
 
\begin{array}{l}
 \text{According }\text{to }\text{the }\text{above }\text{equation}: \\
 v\text{ = }\left(
\begin{array}{c}
 v_1 \\
 v_2 \\
\end{array}
\right)\text{ = }\left(
\begin{array}{c}
 -\frac{1}{10} \left(2+i \sqrt{46}\right) v_2 \\
 v_2 \\
\end{array}
\right) \\
\end{array}
 \\
 
\begin{array}{l}
 \text{Letting }v_2=1 \text{in }\left(
\begin{array}{c}
 -\frac{1}{10} \left(2+i \sqrt{46}\right) v_2 \\
 v_2 \\
\end{array}
\right), \text{we }\text{find }\text{that }\left(
\begin{array}{c}
 \frac{1}{10} \left(-i \sqrt{46}-2\right) \\
 1 \\
\end{array}
\right) \text{is }\text{an }\text{eigenvector }\text{of }\text{the }\text{matrix }\left(
\begin{array}{cc}
 7 & 5 \\

\end{array}
\right) \text{associated }\text{with }\text{the }\text{eigenvalue }5+i \sqrt{46}: \\
 v=\left(
\begin{array}{c}
 \frac{1}{10} \left(-i \sqrt{46}-2\right) \\

\right) \\
\end{array}
